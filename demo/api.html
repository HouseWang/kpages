Co.CustomDict 自定义字典

var dict ={};
Co.CustomDict(dict,'domain','barfoo.com'); //设置dict['domain']='barfoo.com';
Co.CustomDict(dict,{
    'domain':'barfoo.com',
    'nickname':'comger'
});//当key为字典时，dict = $.extend(dict,key) 将 key的键值赋给dict,此时 dict 添加或重写了 domain,nickname
Co.CustomDict(dict,'nickname')  //当key为字符串时，返回dict[nickname] ('comger')

Filter(fn,arr)

过滤 返回满足 fn 的，arr 项数组

Map(fn,arr)

映射 map函数fn 作用于给定序列的每个元素，并用一个列表来提供返回值。

var arr = [1,5,6,8,9]
var rs = Co.Map(function(seg){
    return seg%2;
},arr);

Reduce(fn,arr,initail)

化简 fn有两参数，将fn作用于arr 数组的元素，每次携带一对（先前的结果以及下一个序列的元素），
连续的将现有的结果和下一个值作用在获得的随后的结果上，最后减少我们的序列为一个单一的返回值
initail 为计算的初始值，可不传入这参数 

var arr = [1,5,6,8,9]
var rs = Co.Reduce(function(a,b){
    return a+b;
},arr,10); //rs =39

Weibo.Common.Dates

Co.Dates.Convert(d) 转指定参数转成日期对象 ，可接受参数为数组，日期，整数，字符串，对象 

var d = Co.Dates.Convert('2011-02-03'); //将字符串格式的日期转成日期对象
d= Co.Dates.Convert([2011,02,03]);      //将数组格式的日期转成日期对象
d = Co.Dates.Convert(new Date());       //将日期格式的日期转成日期对象
d = Co.Dates.Convert(12354755477);      //将整数格式的日期转成日期对象

Co.Dates.Compare(d1,d2) a,b两时间对比,时间格式出错时返回 NaN
如果 d1在d2前，-1
如果 d1==d2, 0
如果 d1在d2 之后 ，1 

var d1 = new Date();
var d2 = new Date(2001,12,03);
var flag = Co.Dates.Compare(d1,d2)； // flag= 1

Co.Dates.InRange(d,start,end) 检查时间是否在指定范围， 时间格式出错时返回 NaN 

var d = new Date(2011,7,18);
var start = new Date(2010,12,3);
var end = new Date(2011,12,3);
var flag = Co.Dates.InRange(d,start,end)  // flag = true

Function 方法类扩展方

Param() 获取方法参数

     alert(arg+this.num);
}

Bind(arguments) 用另一个函数来封装原始的函数对象，将函数运行时的作用域限定为 thisObj 所指定的对象。
译注：该方法返回一个新的函数对象，返回的函数是原函数将其 this 关键字指向 thisObj 后的版本，并且，如果指定了可选的 arg 参数，还会将原函数中的部分或全部参数固定为 args 指定的值。该方法不会修改原函数对象。 

function add(arg){
     alert(arg+this.num);
}
function demo(){
    this.num =2;
}
function testBind(){
    var d =  demo();  // 
    var fnadd = add.Bind(d,3);
    fnadd(); // 5
    fnadd(6); // 结果还是5 
    add(6)； // 结果为 8

    fnadd = add.Bind(d);
    fnadd(); // error
    fnadd(3); // 5
    add(5); // 7
}

BindAsEventListener()

一个专用于事件的 Bind 方法的变体， 它确保事件触发后会将当前事件对象作为第一个参数传递给事件处理函数。

var obj = { name: 'A nice demo' }; 
function handler(e) { 
	var tag = Event.element(e).tagName.toLowerCase(); 
	var data = Array.prototype.slice.call(arguments); 
	data.shift();
	alert(this.name + '\nClick on a ' + tag + '\nOther args: ' + data.join(', ')); 
}

$(function(){
    $("body").click(function(){
        handler.BindAsEventListener(obj, 1, 2, 3)
        // 现在，页面上的任意点击都会显示 obj.name、所点击元素的小写的标签名称以及 "1, 2, 3" 
    })
})

客户端内存数据管理器 Weibo.StaticObjs as window.So

So.Config(key,val)

为设置简单的键值，val只能是string ;否则不能固化到cookie,失效时间为 So.Expre,默认为一天 

So.Config("domain","barfoo.com");

So.Config(object)

object 为一级字典；同时设置字典的键值。同样值只能使用string 

So.Config({
    domain:"barfoo.com",
    nickname:'comger'
});

So.Config(key)

获取key 对应的配置值 

var doamin = So.Config("domain")

So.Dict(key,val)

设置字典或按key获取字典值,用法与 So.Config 一致，但值可以接受对象

So.Dict("domain","barfoo.com");
So.Dict("Config",{
    widht:"100",
    height:"100"
});
So.Dict("Config") //返回Config 字典对象
