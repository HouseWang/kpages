图形类库清单

    //矩型 Kpages.Graphic.Rect
    opts= {
        x,
        y,
        w,
        h,
        styles:{
            fillStyle,
            strokeStyle,
            lineWidth
        }
    }

    //线条 Kpages.Graphic.Line
    opts = {
        x,
        y,
        endx,
        endy
    }

    //圆型 Kpages.Graphic.Arc
    opts={
        x:100,圆点x坐标
        y:100, 圆点y坐标
        r:50,半径
        w：0

    //多边型 Kpages.Graphic.Fences
    opts = {

        }
